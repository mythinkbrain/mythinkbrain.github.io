Jawaban:

diketahui :

Pt = 12

Pn = 2k

H = 2k-4

H + 2 = 12

ditanya : K = ?

jawab :

H + 2 = 12

(2k-4) + 2 = 12

2k - 4 = 12 - 2

2k - 4 = 10

2k = 10 + 4

2k = 14

K = 14/2

K = 7

pembuktian :

pt = 12 ✓

Pn = 2x7 = 14 ✓

H = pn - 4 = 14 - 4 = 10 ✓

H + 2 = 10 + 2 = 12 ✓

ket :

pt = putih

pn = pink

H = hitam

k = kuning
