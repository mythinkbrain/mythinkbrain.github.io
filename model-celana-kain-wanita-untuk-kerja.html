dik.
c=celana
b=baju
5c+4b=250.000
4c+2b=170.000

dit.
2c+2b=?

jawab

4c + 2b = 170.000 (kali 2)

5c + 4b = 250.000
8c + 4b = 340.000
-------------------------- -
-3c = -90.000

3c = 90.000
c = Rp.30.000,00

4c + 2b = 170.000
4(30.000) + 2b = 170.000
120.000 + 2b = 170.000
2b = 170.000 - 120.000
2b = 50.000
b = Rp.25.000,00

2c + 2b
= 2(30.000) + 2(25.000)
= 60.000 + 50.000
= Rp.110.000,00
